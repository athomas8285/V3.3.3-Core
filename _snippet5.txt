function __expandTargetMatch(){
  var tm = window.__TARGET_MATCH;
  if(!tm) return;
  delete window.__TARGET_MATCH;
  var cards = document.querySelectorAll(".ana-card");
  for(var i = 0; i < cards.length; i++){
    var c = cards[i];
    var home = c.querySelector(".tname.home");
    var away = c.querySelector(".tname.away");
    if(home && away && home.textContent === tm.h && away.textContent === tm.a){
      toggleAnaCard(c);
      setTimeout(function(){
        c.scrollIntoView({behavior:"smooth",block:"center"});
        c.classList.add("highlight");
        setTimeout(function(){ c.classList.remove("highlight"); }, 1500);
      }, 100);
      break;
    }
  }
}
